#include "scicos_block4.h"

extern int C2F(dprxc)();
void root_coef(scicos_block *block,int flag)

  double *y;
  int mu;
  mu=GetInPortRows(block,1);

  u=GetRealInPortPtrs(block,1);

  C2F(dprxc)(&mu,u,y);
